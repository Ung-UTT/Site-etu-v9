def ldap_attribute_for_username(ldap_uid, attr)
  ldap_uid.downcase!
  @@ldap.search(:base => "uid=#{ldap_uid},ou=people,dc=utt,dc=fr", :attributes => ["#{attr}"]) do |e|
    val = e.send("#{attr}")
    return unpack(val)
  end
end

    return flatten_entry(entry)
