def self.authenticate(login, password)
  user = find_by_login(login)
  if user && user.crypted_password == BCrypt::Engine.hash_secret(password, user.password_salt)
    user
  else
    nil
  end
end

def is?(name, asso = nil)
  res = roles.select { |r| r.symbol == name }
  if asso
    res = res & asso.roles
  end
  !res.empty?
end
